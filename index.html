<script>

  const tickerContent = document.getElementById('ticker-content');


  async function fetchSportsNews() {

    try {

      const res = await fetch('https://api.rss2json.com/v1/api.json?rss_url=https://www.espn.com/espn/rss/news');

      const data = await res.json();

      if (data.status !== 'ok') throw new Error('Failed to fetch RSS');


      const headlines = data.items.slice(0, 15).map(item => {

        return `⚽️ ${item.title}  ⚾️  🏀  🏈  — `;

      }).join(' ');


      tickerContent.textContent = headlines;

    } catch (e) {

      tickerContent.textContent = 'Failed to load sports news.';

      console.error(e);

    }

  }


  fetchSportsNews();

  setInterval(fetchSportsNews, 5 * 60 * 1000);

</script>
